题意: nnn个节点的树，每个节点iii有权值aia_ia​i​​，从111号节点出发求和aia_ia​i​​，当sum<0sum<0sum<0时sumsumsum会自动变为000，可以从树上任意节点结束。求summaxsum_{max}sum​max​​

分析: 我们可以这样考虑每个节点：从该节点以后会存在sumsumsum归零情况或不会。分别设为F[i][0]F[i][0]F[i][0]和F[i][1]F[i][1]F[i][1]

  • 若考虑sumsumsum不会归零，dfs里每一节点F[i][1]F[i][1]F[i][1]都加上max{0,F[j][1]}max\,\{0,F[j][1]\}max{0,F[j][1]},(j取i儿子)；

    F[u][1] += max(F[v][1], 0LL);//F[u][1]初始为w[u]
    1

  • 若考虑sumsumsum会归零，我们可以先将该点的值F[i][0]F[i][0]F[i][0]设为000，用000把后面一堆负数全部提前解决了再处理。

    F[u][0] += max(F[v][0], F[v][1]);//F[u][0]初始为0
    1

const int N = 5e5 + 10;
int n, m;
int flag;
int vis[N], vis2[N];
ll F[N][2], w[N];
vector<int>e[N];
void dfs(int u) {
	if (vis[u])
		return;
	vis[u] = 1;
	F[u][1] = w[u];
	for (auto v : e[u]) {
		if (!vis[v]) {
			dfs(v);
			F[u][1] += max(F[v][1], 0LL);//不会变成0，直接求和
			F[u][0] += max(F[v][0], F[v][1]);//假设后面会变成0，也就是会有一段负的，F[u][0]初始为0把所有负的干了
		}
	}
}

	}
	for (int i = 1; i <= n; i++) scanf("%lld", &w[i]);
	dfs(1);
	printf("%lld", max(F[1][0],F[1][1]));
}
